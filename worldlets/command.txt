!command/help!
parent = "generic/char_command"
name = "help"
category = "Information"

def run(character):
    commands = !menu/game!.get_commands(character)
    categories = {
        "General": [],
        "Information": [],
        "Administrator": [],
    }

    for command in commands:
        category = command.category
        if category in categories:
            categories[category].append(command.name)
        endif
    done

    hlist = display.HorizontalList(columns=4, col_width=15)
    for category in categories.keys():
        commands = categories[category]
        if commands:
            hgroup = hlist.add_group(category)
            for command in commands:
                hgroup.add_entry(command)
            done
        endif
    done

    character.msg(hlist.format())

!command/quit!
parent = "generic/char_command"
name = "quit"
syntax = ""
category = "General"

def run(character):
    log(str(character))
    character.former_location = character.location
    character.location = None
    client = clients.owning(character)

    client.disconnect()
