    account = client.account

        character = account.character

!menu/new/username!
parent = "generic/menu"
text = """
Welcome, new user! Enter your new username.
"""

def input(client, input):
    username = input.strip().lower()
    account = search.one(!account!, username=username)
    if not username:
        client.msg("This username isn't valid. Choose another one.")
    elif account:
        client.msg("This account already exists. Try a new one.")
    else:
        account = Entity(parent=!account!)
        account.username = username
        client.account = account
        client.location = !menu/new/password!
    endif
